
t = -20:20;

a = exp(-(t.^2));
b = t.*(t >= 0);

u = conv(a, b);

subplot(3, 1, 1), stem(t, a); grid on;
xlabel('Time'); ylabel('Amplitude');
title('Input:a(t) = e^{-t^2}');

subplot(3, 1, 2), stem(t, b); grid on;
xlabel('Time'); ylabel('Amplitude');
title('Input:b(t) = t, for t >= 0');

n = (2*t(1)):(2*t(end));

subplot(3, 1, 3), stem(n, u); grid on;
xlabel('Time'); ylabel('Amplitude');
title('u(t):a(t)*b(t)');
